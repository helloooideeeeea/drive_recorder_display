サウンドカードデバイスネーム確認
cat /proc/asound/cards
arecord -l

サウンドカードデバイスバス確認
find /sys/devices/ -name id | grep sound

カメラのSYSFSを確認
ll /sys/class/video4linux

udevテスト udevで使う値が取得できる
udevadm test -a add $(udevadm info -q path -n /dev/video2)
udevadm info --query=all --name=/dev/video0


カメラリスト
v4l2-ctl --list-devices

対応している色表現フォーマット
v4l2-ctl -d /dev/video-inside --list-formats

対応している解像度等
v4l2-ctl -d /dev/video-inside --list-formats-ext

解像度変更等
v4l2-ctl -d /dev/video-inside -v width=1920,height=1080,pixelformat=MJPG
v4l2-ctl -d /dev/video-outside -v width=1920,height=1080,pixelformat=MJPG

フレームレート
v4l2-ctl -d /dev/video-inside -p 25
v4l2-ctl -d /dev/video-outside -p 30

詳細
v4l2-ctl -d /dev/video-inside --all

仮想カメラデバイスを登録(不採用)
sudo modprobe v4l2loopback

複数の仮想カメラデバイスをつくる。
# video66とvideo88は、録音専用デバイス。video66は、内側カメラ。video88は、外側カメラ
# video77とvideo99は、プログラムアクセス用。video77は、内側カメラ。video99は、外側カメラ
sudo modprobe v4l2loopback video_nr=66,77,88,99

仮想カメラデバイスを削除
sudo modprobe -r v4l2loopback

カメラデバイスのRGBを仮想カメラデバイスに送る（物理デバイスは、マルチプロセスをブロックするため。不採用）
gst-launch-1.0 v4l2src device=/dev/video-inside ! image/jpeg,width=1920,height=1080,framerate=25/1 ! jpegdec ! videoconvert ! video/x-raw,format=I420,width=1920,height=1080,framerate=25/1 ! v4l2sink device=/dev/video66

マルチで送る場合（こっちを採用）
gst-launch-1.0 v4l2src device=/dev/video-inside ! image/jpeg,width=1280,height=720,framerate=30/1 ! jpegdec ! videoconvert ! video/x-raw,format=I420,width=1280,height=720,framerate=30/1 ! tee name=tp tp. ! queue ! v4l2sink device=/dev/video66 tp. ! queue ! v4l2sink device=/dev/video77
gst-launch-1.0 v4l2src device=/dev/video-outside ! image/jpeg,width=1280,height=720,framerate=30/1 ! jpegdec ! videoconvert ! video/x-raw,format=I420,width=1280,height=720,framerate=30/1 ! tee name=tp tp. ! queue ! v4l2sink device=/dev/video88 tp. ! queue ! v4l2sink device=/dev/video99

完成系（動画音声ミックス。ハードウェアエンコード。）
gst-launch-1.0 -e v4l2src device="/dev/video66" ! queue ! video/x-raw, width=1280, height=720 ! nvvidconv ! nvv4l2h264enc ! h264parse ! mux. alsasrc device="hw:audio-inside,0" ! queue ! audioconvert ! voaacenc ! mpegtsmux name=mux ! hlssink max-files=0 target-duration=60 location=./hls/segment%05d.ts playlist-location=./hls/playlist.m3u8
